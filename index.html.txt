var N = slides.length;
var cur = 0;

// assign images with shuffle
var pool = imgs.slice();

// dots
var de = document.getElementById('dots');
for(var i=0;i<N;i++){
  var d=document.createElement('div');
  d.className='dot'+(i===0?' on':'');
  de.appendChild(d);
}

function go(n){
  slides[cur].classList.remove('on');
  de.children[cur].classList.remove('on');
  cur=(n+N)%N;
  slides[cur].classList.add('on');
  de.children[cur].classList.add('on');
}

// tap navigation
slides.forEach(function(s){
  s.querySelector('.tap-up').addEventListener('click',function(e){e.stopPropagation();go(cur-1);});
  s.querySelector('.tap-dn').addEventListener('click',function(e){e.stopPropagation();go(cur+1);});
});

// swipe
var sy=0;
document.getElementById('app').addEventListener('touchstart',function(e){sy=e.touches[0].clientY;},{passive:true});
document.getElementById('app').addEventListener('touchend',function(e){
  var dy=sy-e.changedTouches[0].clientY;
  if(dy>50)go(cur+1);
  else if(dy<-50)go(cur-1);
},{passive:true});

// like

document.querySelectorAll('.sbtn').forEach(function(b){
  b.addEventListener('click',function(e){
    e.stopPropagation();
    var u=b.closest('.slide').getAttribute('data-img');
    if(navigator.share)navigator.share({url:u});
    else if(navigator.clipboard)navigator.clipboard.writeText(u);
  });
});

// download
document.querySelectorAll('.dbtn').forEach(function(b){

    e.stopPropagation();

    var a=document.createElement('a');
    a.href=u;a.download='image.jpg';a.target='_blank';
    document.body.appendChild(a);a.click();a.remove();
  });
